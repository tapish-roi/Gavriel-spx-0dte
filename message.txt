<script>
(function(){
  'use strict';
  var reduceMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

  var canvas = document.getElementById('waves');
  var ctx = canvas.getContext('2d');
  var W = 0, H = 0;

  function resize(){
    var dpr = Math.min(window.devicePixelRatio || 1, 2);
    W = window.innerWidth;
    H = window.innerHeight;
    canvas.width = Math.round(W * dpr);
    canvas.height = Math.round(H * dpr);
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  }
  resize();

  var layers = [
    { base: 0.16, amp: 34, wl: 0.0036, sp: 0.10, amp2: 18, wl2: 0.0013, sp2: 0.05,  col: '225, 64, 90',   a: 0.17, fill: 'up'   },
    { base: 0.30, amp: 46, wl: 0.0028, sp: 0.14, amp2: 26, wl2: 0.0009, sp2: 0.07,  col: '16, 185, 129',  a: 0.17, fill: 'up'   },
    { base: 0.62, amp: 52, wl: 0.0031, sp: 0.12, amp2: 30, wl2: 0.0011, sp2: 0.06,  col: '16, 185, 129',  a: 0.21, fill: 'down' },
    { base: 0.72, amp: 44, wl: 0.0042, sp: 0.17, amp2: 24, wl2: 0.0014, sp2: 0.09,  col: '52, 211, 153',  a: 0.24, fill: 'down' },
    { base: 0.82, amp: 38, wl: 0.0052, sp: 0.22, amp2: 20, wl2: 0.0017, sp2: 0.11,  col: '225, 64, 90',   a: 0.18, fill: 'down' },
    { base: 0.90, amp: 26, wl: 0.0064, sp: 0.28, amp2: 14, wl2: 0.0021, sp2: 0.14,  col: '245, 239, 224', a: 0.11, fill: 'down' }
  ];

  function waveY(L, x, t){
    return H * L.base
      + Math.sin(x * L.wl + t * L.sp * Math.PI * 2) * L.amp
      + Math.sin(x * L.wl2 - t * L.sp2 * Math.PI * 2 + 1.7) * L.amp2
      + Math.sin(t * 0.21 + L.base * 9) * 9;
  }

  function drawLayer(L, t){
    var step = 8;
    ctx.beginPath();
    ctx.moveTo(-step, waveY(L, -step, t));
    for (var x = 0; x <= W + step; x += step){
      ctx.lineTo(x, waveY(L, x, t));
    }
    ctx.strokeStyle = 'rgba(' + L.col + ', ' + (L.a * 2.6).toFixed(3) + ')';
    ctx.lineWidth = 1.6;
    ctx.shadowColor = 'rgba(' + L.col + ', 0.55)';
    ctx.shadowBlur = reduceMotion ? 0 : 16;
    ctx.stroke();
    ctx.shadowBlur = 0;

    var yTop = H * L.base - L.amp - L.amp2 - 12;
    var yBot = H * L.base + L.amp + L.amp2 + 12;
    var g;
    if (L.fill === 'down'){
      ctx.lineTo(W + step, H + 10);
      ctx.lineTo(-step, H + 10);
      g = ctx.createLinearGradient(0, yTop, 0, H);
      g.addColorStop(0, 'rgba(' + L.col + ', ' + L.a.toFixed(3) + ')');
      g.addColorStop(1, 'rgba(' + L.col + ', 0)');
    } else {
      ctx.lineTo(W + step, -10);
      ctx.lineTo(-step, -10);
      g = ctx.createLinearGradient(0, yBot, 0, 0);
      g.addColorStop(0, 'rgba(' + L.col + ', ' + L.a.toFixed(3) + ')');
      g.addColorStop(1, 'rgba(' + L.col + ', 0)');
    }
    ctx.closePath();
    ctx.fillStyle = g;
    ctx.fill();
  }

  var dust = [];
  for (var d = 0; d < 42; d++){
    dust.push({
      x: Math.random(), y: Math.random(),
      r: 0.6 + Math.random() * 1.5,
      sp: 0.004 + Math.random() * 0.01,
      ph: Math.random() * Math.PI * 2
    });
  }
  function drawDust(t){
    for (var i = 0; i < dust.length; i++){
      var p = dust[i];
      var px = (p.x + t * p.sp) % 1;
      var py = p.y + Math.sin(t * 0.4 + p.ph) * 0.012;
      var tw = 0.35 + 0.65 * (0.5 + 0.5 * Math.sin(t * 1.4 + p.ph * 3));
      ctx.fillStyle = 'rgba(245, 239, 224, ' + (0.16 * tw).toFixed(3) + ')';
      ctx.beginPath();
      ctx.arc(px * W, py * H, p.r, 0, Math.PI * 2);
      ctx.fill();
    }
  }

  var grainTile = document.createElement('canvas');
  grainTile.width = grainTile.height = 128;
  (function(){
    var gctx = grainTile.getContext('2d');
    var img = gctx.createImageData(128, 128);
    for (var i = 0; i < img.data.length; i += 4){
      var v = (Math.random() * 255) | 0;
      img.data[i] = img.data[i + 1] = img.data[i + 2] = v;
      img.data[i + 3] = 255;
    }
    gctx.putImageData(img, 0, 0);
  })();
  var grainPattern = ctx.createPattern(grainTile, 'repeat');

  function drawFrame(now){
    var t = now / 1000;
    ctx.clearRect(0, 0, W, H);

    var glow = ctx.createRadialGradient(W / 2, H * 0.55, 0, W / 2, H * 0.55, Math.max(W, H) * 0.6);
    glow.addColorStop(0, 'rgba(16, 185, 129, 0.05)');
    glow.addColorStop(1, 'rgba(16, 185, 129, 0)');
    ctx.fillStyle = glow;
    ctx.fillRect(0, 0, W, H);

    for (var i = 0; i < layers.length; i++) drawLayer(layers[i], reduceMotion ? 0 : t);
    if (!reduceMotion) drawDust(t);

    if (!reduceMotion){
      var cycle = (t % 9) / 9;
      var sx = cycle * (W * 1.8) - W * 0.4;
      ctx.save();
      ctx.translate(sx, H / 2);
      ctx.rotate(-0.35);
      var sw = 200;
      var sg = ctx.createLinearGradient(-sw, 0, sw, 0);
      sg.addColorStop(0, 'rgba(245, 239, 224, 0)');
      sg.addColorStop(0.5, 'rgba(245, 239, 224, 0.04)');
      sg.addColorStop(1, 'rgba(245, 239, 224, 0)');
      ctx.fillStyle = sg;
      ctx.fillRect(-sw, -H * 1.2, sw * 2, H * 2.4);
      ctx.restore();
    }

    var vg = ctx.createRadialGradient(W / 2, H / 2, Math.min(W, H) * 0.35, W / 2, H / 2, Math.max(W, H) * 0.75);
    vg.addColorStop(0, 'rgba(5, 8, 7, 0)');
    vg.addColorStop(1, 'rgba(5, 8, 7, 0.6)');
    ctx.fillStyle = vg;
    ctx.fillRect(0, 0, W, H);

    if (grainPattern){
      ctx.save();
      ctx.globalAlpha = 0.05;
      var ox = reduceMotion ? 0 : (Math.random() * 128) | 0;
      var oy = reduceMotion ? 0 : (Math.random() * 128) | 0;
      ctx.translate(-ox, -oy);
      ctx.fillStyle = grainPattern;
      ctx.fillRect(0, 0, W + 128, H + 128);
      ctx.restore();
    }
  }

  var rafId = null;
  function loop(now){
    drawFrame(now);
    rafId = requestAnimationFrame(loop);
  }
  drawFrame(performance.now());
  window.addEventListener('resize', function(){ resize(); drawFrame(performance.now()); });

  if (!reduceMotion){
    rafId = requestAnimationFrame(loop);
    document.addEventListener('visibilitychange', function(){
      if (document.hidden){
        if (rafId){ cancelAnimationFrame(rafId); rafId = null; }
      } else if (!rafId){
        rafId = requestAnimationFrame(loop);
      }
    });
  }

  var form = document.getElementById('leadForm');
  var btn = document.getElementById('submitBtn');
  var msg = document.getElementById('formMsg');
  var btnHTML = btn.innerHTML;

  function setErr(id, show){
    var input = document.getElementById(id);
    var err = document.getElementById('err-' + id);
    input.classList.toggle('invalid', show);
    err.classList.toggle('show', show);
    if (show) input.setAttribute('aria-invalid', 'true');
    else input.removeAttribute('aria-invalid');
  }

  function validate(){
    var ok = true;
    var name = document.getElementById('fullname').value.trim();
    var bad = name.length < 2;
    setErr('fullname', bad); if (bad) ok = false;

    var age = parseInt(document.getElementById('age').value, 10);
    bad = !(age >= 16 && age <= 99);
    setErr('age', bad); if (bad) ok = false;

    var phone = document.getElementById('phone').value.replace(/[\s\-()]/g, '');
    bad = !/^0(5\d{8}|[2-489]\d{7})$/.test(phone);
    setErr('phone', bad); if (bad) ok = false;

    var email = document.getElementById('email').value.trim();
    bad = !/^[^\s@]+@[^\s@]+\.[^\s@]{2,}$/.test(email);
    setErr('email', bad); if (bad) ok = false;

    return ok;
  }

  ['fullname', 'age', 'phone', 'email'].forEach(function(id){
    document.getElementById(id).addEventListener('input', function(){ setErr(id, false); });
  });

  form.addEventListener('submit', function(ev){
    ev.preventDefault();
    msg.className = 'form-msg';
    msg.textContent = '';
    if (!validate()){
      var firstBad = form.querySelector('.invalid');
      if (firstBad) firstBad.focus();
      return;
    }

    btn.disabled = true;
    btn.textContent = 'שולח...';

    var endpoint = form.action.replace('formsubmit.co/', 'formsubmit.co/ajax/');
    var data = new FormData(form);

    fetch(endpoint, {
      method: 'POST',
      body: data,
      headers: { 'Accept': 'application/json' }
    }).then(function(res){
      if (!res.ok) throw new Error('HTTP ' + res.status);
      return res.json();
    }).then(function(){
      form.querySelectorAll('.fields, button[type="submit"]').forEach(function(el){ el.style.display = 'none'; });
      msg.className = 'form-msg success';
      msg.textContent = 'הפרטים נשלחו בהצלחה! גבריאל ייצור איתך קשר בקרוב.';
    }).catch(function(){
      btn.disabled = false;
      btn.innerHTML = btnHTML;
      msg.className = 'form-msg error';
      msg.textContent = 'משהו השתבש בשליחה. נסה שוב בעוד רגע, או רענן את הדף.';
    });
  });
})();
